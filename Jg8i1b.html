package network; 
import java.io.; 
import java.io.; 
import java.io.;

public class Network  
{ 
	protected int serverPort; 
	protected  serverSocket;  
	protected  socket;  
	 
	private  in; 
	private  out;  
	private  waitForCounterpart;  
 
	public Network() 
	{	 
		serverPort = 7700; 
		serverSocket = null; 
		socket = null; 
		in = null; 
		out = null; 
		waitForCounterpart = null; 
	} 
	 
	public void connectAsServer(  obj )   
	{ 
		try  
		{ 
			serverSocket = new ( serverPort );  
			socket =  
			         serverSocket.();    
			in = new ( new ( socket.() ) ); 
			out = new ( socket.(), true ); 
			waitForCounterpart = new ( obj );  
			waitForCounterpart.();   
		} 
		catch ( Exception e ) 
		{ 
			e.printStackTrace(); 
		} 
	}  
 
	public void connectAsClient( String serverIP,  obj )   
	{ 
		try  
		{ 
			socket =  
			         new ( serverIP, serverPort );  
			in = new ( new ( socket.() )); 
			out = new ( socket.(), true ); 
			waitForCounterpart = new ( obj );  
			waitForCounterpart.();	 
		}  
		catch ( Exception e ) 
		{ 
			e.printStackTrace(); 
		} 
	} 
	 
	public String read()  
	{ 
		try  
		{ 
			if ( this.isConnecting() == true ) 
				return  
				       in.readLine();
		}  
		catch ( Exception e ) 
		{ 
			e.printStackTrace(); 
		} 
		 
		return null; 
	} 
	 
	public void write( String data )  
	{ 
		try  
		{ 
			if ( this.isConnecting() == true ) 
				out.println( data ); 
		}  
		catch ( Exception e ) 
		{ 
			e.printStackTrace(); 
		}	 
	} 
	 
	public void disconnect()  
	{ 
		try  
		{ 
			if( waitForCounterpart != null ) 
			{ 
				waitForCounterpart.();	   
				waitForCounterpart = null; 
			} 
			if( in != null ) 
			{ 
				in.(); 
				in=null; 
			}			 
			if( out != null ) 
			{ 
				out.(); 
				out.(); 
				out = null; 
			} 
			if( socket != null ) 
			{ 
				socket.(); 
				socket = null; 
			} 
			if( serverSocket != null ) 
			{ 
				serverSocket.(); 
				serverSocket = null; 
			} 
		}  
		catch ( Exception e ) 
		{ 
			e.printStackTrace(); 
		} 
	} 
 
	public boolean isConnecting()  
	{ 
		if ( ( socket != null )  
				&& ( in != null )  
				&& ( out != null )  
				&& ( waitForCounterpart != null )  
				&& ( waitForCounterpart.getState() != .State.TERMINATED ) )    
			return true; 
		else 
			return false; 
	}
